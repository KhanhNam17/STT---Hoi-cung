# requirements-sortformer.txt

# MÔI TRƯỜNG RIÊNG cho NVIDIA Streaming Sortformer (NeMo).
# KHÔNG cài chung với stt_uc2_env! NeMo kéo torch mới (vd 2.12) xung đột với
# torch==2.2.2 mà app (diart/pyannote/qai_hub) cần → vỡ cả 2.
#
# Tạo env riêng rồi cài file này:
#   conda create -n sortformer_env python=3.10 -y
#   conda activate sortformer_env
#   pip install -r requirements-sortformer.txt
#
# ⚠️ NeMo yêu cầu Cython + packaging TRƯỚC. Nếu lệnh trên lỗi, cài 2 bước:
#   pip install Cython packaging
#   pip install "nemo_toolkit[asr]"
#   (bản pypi lỗi thì thử main:)
#   pip install "git+https://github.com/NVIDIA/NeMo.git@main#egg=nemo_toolkit[asr]"
#
# Sau khi cài, diarize 1 file (16kHz mono) trong env này:
#   python core/diarization/sortformer.py path\to\file_16k.wav 2
#
# NeMo tự kéo torch / torchvision / torchaudio khớp nhau + numpy.
# ─────────────────────────────────────────────────────────────────────────────
